# SASS - Simple Alpine SendSpin - boot config
# Shared config for Raspberry Pi 3 and 4

# Enable audio (onboard + required for USB audio class detection)
dtparam=audio=on

# Give the GPU minimal memory, headless audio appliance
gpu_mem=16

# Boot faster
disable_splash=1
boot_delay=0

[pi3]
kernel=vmlinuz-rpi
initramfs initramfs-rpi

[pi4]
kernel=vmlinuz-rpi
initramfs initramfs-rpi
arm_boost=1

[all]
arm_64bit=1
# Physically enable the primary UART (needed for the serial console already
# requested in cmdline.txt) so boot failures can be diagnosed over serial.
enable_uart=1
